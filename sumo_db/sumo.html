Types

condition() =
            {'and', [condition()]} |
            {'or', [condition()]} |
            {'not', condition()} |
            {field_name(), field_value()} |
            {field_name(), operator(), field_value()} |
            {field_name(), operator(), field_name()}

conditions() = condition() | [condition()]

doc() = #{field_name() => field_value()}

field() = sumo_internal:field()

field_attr() =
            id |
            unique |
            index |
            not_null |
            auto_increment |
            {length, integer()}

field_attrs() = [field_attr()]

field_name() = atom()

field_type() =
            integer | string | binary | text | float | date | datetime

field_value() = term()

operator() = '<' | '>' | '==' | '=<' | '>=' | '/=' | like

schema() = sumo_internal:schema()

schema_name() = atom()

sort() = field_name() | [{field_name(), sort_order()}]

sort_order() = asc | desc

user_doc() = term()
